    <script>

        // Water animation and ripple effect

        const canvas = document.getElementById('waterCanvas');

        const ctx = canvas.getContext('2d');

        

        // Set canvas to full window size

        function resizeCanvas() {

            canvas.width = window.innerWidth;

            canvas.height = window.innerHeight;

        }

        

        window.addEventListener('resize', resizeCanvas);

        resizeCanvas();

        

        // Generate random water color on page load

        function getRandomColor() {

            const hue = Math.floor(Math.random() * 360);

            const saturation = 60 + Math.floor(Math.random() * 40); // 60-100%

            const lightness = 30 + Math.floor(Math.random() * 40); // 30-70%

            return `hsl(${hue}, ${saturation}%, ${lightness}%)`;

        }

        

        const waterColor = getRandomColor();

        document.documentElement.style.setProperty('--water-color', waterColor);

        

        // Determine text color based on background brightness

        function getBrightness(color) {

            // For HSL colors

            const match = color.match(/hsl\((\d+),\s*(\d+)%,\s*(\d+)%\)/);

            if (match) {

                const lightness = parseInt(match[3]);

                return lightness;

            }

            return 50; // Default to middle lightness if parsing fails

        }

        

        const brightness = getBrightness(waterColor);

        const textColor = brightness > 50 ? 'black' : 'white';

        document.documentElement.style.setProperty('--text-color', textColor);

        

        // Water wave simulation

        const waves = [];

        const ripples = [];

        

        class Wave {

            constructor() {

                this.x = 0;

                this.amplitude = Math.random() * 20 + 5;

                this.period = Math.random() * 200 + 100;

                this.speed = Math.random() * 0.05 + 0.01;

                this.phase = Math.random() * Math.PI * 2;

                this.color = waterColor;

            }

            

            update() {

                this.phase += this.speed;

                if (this.phase > Math.PI * 2) {

                    this.phase -= Math.PI * 2;

                }

            }

            

            draw(ctx) {

                ctx.beginPath();

                ctx.moveTo(0, canvas.height / 2);

                

                for (let x = 0; x < canvas.width; x += 5) {

                    const y = Math.sin(x / this.period + this.phase) * this.amplitude;

                    ctx.lineTo(x, canvas.height / 2 + y);

                }

                

                ctx.lineTo(canvas.width, canvas.height);

                ctx.lineTo(0, canvas.height);

                ctx.closePath();

                

                const gradient = ctx.createLinearGradient(0, 0, 0, canvas.height);

                gradient.addColorStop(0, this.color);

                gradient.addColorStop(1, adjustColor(this.color, -20));

                

                ctx.fillStyle = gradient;

                ctx.fill();

            }

        }

        

        class Ripple {

            constructor(x, y) {

                this.x = x;

                this.y = y;

                this.radius = 0;

                this.maxRadius = Math.random() * 100 + 50;

                this.speed = Math.random() * 2 + 1;

                this.opacity = 1;

                this.color = adjustColor(waterColor, 20);

            }

            

            update() {

                this.radius += this.speed;

                this.opacity = 1 - (this.radius / this.maxRadius);

                return this.radius < this.maxRadius;

            }

            

            draw(ctx) {

                ctx.beginPath();

                ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2);

                ctx.strokeStyle = `${this.color}${Math.floor(this.opacity * 255).toString(16).padStart(2, '0')}`;

                ctx.lineWidth = 2;

                ctx.stroke();

            }

        }

        

        // Function to adjust color brightness

        function adjustColor(color, amount) {

            const match = color.match(/hsl\((\d+),\s*(\d+)%,\s*(\d+)%\)/);

            if (match) {

                const h = parseInt(match[1]);

                const s = parseInt(match[2]);

                let l = parseInt(match[3]) + amount;

                l = Math.max(0, Math.min(100, l));

                return `hsl(${h}, ${s}%, ${l}%)`;

            }

            return color;

        }

        

        // Initialize waves

        for (let i = 0; i < 5; i++) {

            waves.push(new Wave());

        }

        

        // Click event for ripples

        canvas.addEventListener('click', (event) => {

            const rect = canvas.getBoundingClientRect();

            const x = event.clientX - rect.left;

            const y = event.clientY - rect.top;

            ripples.push(new Ripple(x, y));

        });

        

        // Animation loop

        function animate() {

            ctx.clearRect(0, 0, canvas.width, canvas.height);

            

            // Draw base water color

            ctx.fillStyle = waterColor;

            ctx.fillRect(0, 0, canvas.width, canvas.height);

            

            // Update and draw waves

            for (const wave of waves) {

                wave.update();

                wave.draw(ctx);

            }

            

            // Update and draw ripples

            for (let i = ripples.length - 1; i >= 0; i--) {

                const active = ripples[i].update();

                if (active) {

                    ripples[i].draw(ctx);

                } else {

                    ripples.splice(i, 1);

                }

            }

            

            requestAnimationFrame(animate);

        }

        

        animate();

    </script>
